Listes d'association

  Lemma swap_assoc :
    ∀ R R' x x0 t0 t,
      assoc x0 R = None →
      assoc x ((x0,t0) :: R ++ R') = Some t →
      assoc x (R ++ (x0,t0) :: R') = Some t.

  Lemma L0 :
    ∀ R R' x0 t,
      assoc x0 R = None →
      assoc x0 R' = Some t →
      assoc x0 (R ++ R') = Some t.

  Lemma L1 : ∀ R R' x x0 t0 t,
      assoc x0 R = None →
      x ≠ x0 →
      assoc x (R ++ R') = Some t →
      assoc x (R ++ (x0, t0) :: R') = Some t.

  Lemma swap_assoc' :
    ∀ R R' Rr x x0 t0 t,
      assoc x0 R = None →
      assoc x (Rr ++ (x0,t0) :: R ++ R') = Some t →
      assoc x (Rr ++ R ++ (x0,t0) :: R') = Some t.

Lemma swap_mle_r :
  ∀ mle R T t x t' R' Rr ,
    assoc x R = None →
    OCamlEexpr ( Rr ++ ( x , t' ) :: R ++ R' ) T mle t →
    OCamlEexpr ( Rr ++ R ++ ( x, t') :: R' ) T mle t
with swap_mles_r :
       ∀ mles R T t x t' R' Rr ,
         assoc x R = None →
         OCamlEexprs ( Rr ++ ( x , t' ) :: R ++ R' ) T mles t →
         OCamlEexprs ( Rr ++ R ++ ( x, t') :: R' ) T mles t
with swap_decls_r :
       ∀ decls R R' Rr T ts x t',
         assoc x R = None →
       OCamlEdecls (Rr ++ (x, t') :: R ++ R') T decls ts →
       OCamlEdecls (Rr ++ R ++ (x, t') :: R') T decls ts
with swap_declsr_r :
       ∀ declsr R R' Rr T ts x t',
         assoc x R = None →
       OCamlEdeclsr (Rr ++ (x, t') :: R ++ R') T declsr ts →
       OCamlEdeclsr (Rr ++ R ++ (x, t') :: R') T declsr ts.
