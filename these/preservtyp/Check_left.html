Fonctions utiles
Fixpoint type_eqn eqn ( T : T) :=
  match eqn with
  | Equation_exp x _ _ ⇒
    assoc x T
  | Equation_fby x _ _ _ ⇒
    assoc x T
  end.

Fixpoint types_eqns eqns T :=
  match eqns with
    | nil ⇒ Some (TS_nil)
    | cons eqn eqns ⇒
      match (type_eqn eqn T) with
      | Some t ⇒ match (types_eqns eqns T) with
                  | None ⇒ None
                  | Some ts ⇒ Some (TS_cons t ts)
                  end
      | None ⇒ None
      end
  end.

Lemma types_eqns_cons :
  ∀ a eqns T t5 ts5,

    types_eqns eqns T = Some ts5 →
