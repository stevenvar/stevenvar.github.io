Équivalence de typage avant et après traduction

Expressions

Lemma typing_preservation_lexp_equiv :
  ∀ R T le t,
    OCLTyp_t_exp T le t ↔ OCamlEexpr R T (convert_lexp_mle le) t.

Expressions conditionnelles


Lemma typing_preservation_cexp_equiv :
  ∀ R T ce t,
    OCLTyp_t_cexp T ce t ↔ OCamlEexpr R T (convert_cexp_mle ce) t.

Horloges


Lemma typing_preservation_ck_equiv :
 ∀ ck T R,
    OCLt_ck T ck ↔
    OCamlEexpr R T (convert_ck ck) T_bool.

Équations


Lemma typing_preservation_eqns:
    ∀ eqns T R ts',
    Disjoint_disj (names_eqns eqns) →
    OCamlEdeclsr R T (convert_inits_decls eqns) ts' →
    OCamlEexpr R T (convert_updates eqns) T_unit →
    ∀ ts,
      ( OCLt_eqns T eqns ts ↔ OCamlEdecls R T (convert_eqns_decls eqns) ts).

Noeuds


Theorem typing_preservation_node :
  ∀ node t, WF_wf node → nolocals node → ( OCLt_node node t ↔ OCamlEexpr nil nil (convert_node node) (T_arrow (T_types TS_nil) t)).

Noeuds avec flots définis localement

Theorem typing_preservation_node_local_equiv :
  ∀ node t,
    WF_wf (node) →
    (OCL_t_node_locals node t ↔
     OCamlEexpr nil nil (convert_node node) (T_arrow (T_types TS_nil) t)).
