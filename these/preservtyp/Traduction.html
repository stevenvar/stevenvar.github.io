Inductive boolop : Set :=
 | Boolop_and : boolop
 | Boolop_or : boolop.

Definition boolean : Set := bool.

Inductive binop : Set :=
 | Binop_int : binop
 | Binop_bool (boolop5:boolop).

Inductive constant : Set :=
 | K_int : constant
 | K_bool (boolean5:boolean).

Inductive unop : Set :=
 | Unopminus : unop
 | Unopnot : unop .

Inductive lexp : Set :=
 | E_unit : lexp
 | E_const (k:constant)
 | E_var (x:identifier)
 | E_binop (e:lexp) (binop5:binop) (e':lexp)
 | E_unop (unop5:unop) (e:lexp)
 | E_when (e:lexp) (x:identifier)
 | E_whennot (e:lexp) (x:identifier) .

Inductive cexp : Set :=
 | Ce_exp (e:lexp)
 | Ce_merge (x:identifier) (ce:cexp) (ce':cexp)
 | Ce_if (e:lexp) (ce:cexp) (ce':cexp) .

Inductive clock : Set :=
 | Cbase : clock
 | Con (ck:clock) (x:identifier)
 | Connot (ck:clock) (x:identifier) .

Inductive equation : Set :=
 | Equation_exp (x:identifier) (ck:clock) (ce:cexp)
 | Equation_fby (x:identifier) (ck:clock) (k:constant) (e:lexp).

Definition equations : Set := list equation.

Definition xs : Set := list identifier.

Inductive nodedef : Set :=
 | Node_mk (f:identifier) (xs5:xs) (ys:xs) (eqns:equations).

Inductive ts : Set :=
 | TS_nil : ts
 | TS_cons (t5:t) (ts5:ts)
with t : Set :=
 | T_unit : t
 | T_int : t
 | T_bool : t
 | T_types (ts5:ts)
 | T_ref (t5:t)
 | T_arrow (t5:t) (t':t).

Definition program : Set := list nodedef.

Inductive mle : Set :=
 | ML_Emagic : mle
 | ML_Eunit : mle
 | ML_Econst (k:constant)
 | ML_Evariable (x:identifier)
 | ML_Eregister (x:identifier)
 | ML_Eref (mle5:mle)
 | ML_Ederef (mle5:mle)
 | ML_Eupdate (mle5:mle) (mle':mle)
 | ML_Ebinop (mle5:mle) (binop5:binop) (mle':mle)

 | ML_Ebinding_reg (y:identifier) (mle5:mle) (mle':mle)

 | ML_Decls_nil : decls

 | ML_Declsr_nil : declsr
 | ML_Declsr_cons (y:identifier) (mle5:mle) (declsr5:declsr)
with mles : Set :=
 | ML_ESnil : mles
 | ML_EScons (mle5:mle) (mles5:mles).

Fixpoint assoc x (l : T) :=
  match l with
  | nil ⇒ None
  | cons (y,c) l ⇒ if eq_identifier x y then Some c else assoc x l
  end.

 Fixpoint zip (xs : xs) ( ts : ts ) :=
  match xs,ts with
  | cons x xs , TS_cons t ts ⇒
    match zip xs ts with
    | Some xs ⇒ Some ( (x,t)::xs )
    | _ ⇒ None
   end
  | nil, TS_nil ⇒ Some nil
  | _,_ ⇒ None
 end.

Inductive In_ins : xs → xs → Prop :=
 | In_Nil :
     In_ins nil nil
 | In_Cons : ∀ (x:identifier) (xs5 ys:xs),
      (In x ys ) →
     In_ins xs5 ys →
     In_ins (cons x xs5 ) ys.

Inductive Disjoint_disj : xs → Prop :=
 | Disjoint_One_Nil :
     Disjoint_disj nil
 | Disjoint_One_Cons : ∀ (y:identifier) (ys:xs),
      (¬ In y ys ) →
     Disjoint_disj ys →
     Disjoint_disj (cons y ys )
with Disjoint_disjs : xs → xs → Prop :=
 | Disjoint_Two_Nil : ∀ (ys:xs),
     Disjoint_disjs nil ys
 | Disjoint_Two_Cons : ∀ (x:identifier) (xs5 ys:xs),
      (¬ In x ys ) →
     Disjoint_disjs xs5 ys →
     Disjoint_disjs (cons x xs5 ) ys.

Inductive Subin : identifier → t → xs → ts → Prop :=
 | SubIn_True : ∀ (x:identifier) (t5:t) (y:identifier) (ys:xs) (t':t) (ts':ts),
      (ML_Evariable x) = (ML_Evariable y) →
      ( t5 = t' ) →
     Subin x t5 (cons y ys ) (TS_cons t' ts')
 | SubIn_False : ∀ (x:identifier) (t5:t) (y:identifier) (ys:xs) (t':t) (ts':ts),
      ( x ≠ y ) →
     Subin x t5 ys ts' →
     Subin x t5 (cons y ys ) (TS_cons t' ts')
with Subsubset : xs → ts → xs → ts → Prop :=
 | SubSet_Nil : ∀ (ys:xs) (ts':ts),
     Subsubset nil TS_nil ys ts'
 | SubSet_Cons : ∀ (x:identifier) (xs5:xs) (t5:t) (ts5:ts) (ys:xs) (ts':ts),
     Subin x t5 ys ts' →
     Subsubset xs5 ts5 ys ts' →
     Subsubset (cons x xs5 ) (TS_cons t5 ts5) ys ts'.

Inductive OCLTyp_t_exp : T → lexp → t → Prop :=
 | OCLTyp_E_Unit : ∀ (T5:T),
     OCLTyp_t_exp T5 E_unit T_unit
 | OCLTyp_E_Const_i : ∀ (T5:T),
     OCLTyp_t_exp T5 (E_const K_int) T_int
 | OCLTyp_E_Const_b : ∀ (T5:T) (boolean5:boolean),
     OCLTyp_t_exp T5 (E_const (K_bool boolean5)) T_bool
 | OCLTyp_E_Var : ∀ (T5:T) (x:identifier) (t5:t),
      (assoc x T5 = Some t5 ) →
     OCLTyp_t_exp T5 (E_var x) t5
 | OCLTyp_E_IntOp : ∀ (T5:T) (e e':lexp),
     OCLTyp_t_exp T5 e T_int →
     OCLTyp_t_exp T5 e' T_int →
     OCLTyp_t_exp T5 (E_binop e Binop_int e') T_int
 | OCLTyp_E_BoolOp : ∀ (T5:T) (e:lexp) (boolop5:boolop) (e':lexp),
     OCLTyp_t_exp T5 e T_bool →
     OCLTyp_t_exp T5 e' T_bool →
     OCLTyp_t_exp T5 (E_binop e (Binop_bool boolop5) e') T_bool
 | OCLTyp_E_Unop : ∀ (T5:T) (unop5:unop) (e:lexp) (t5:t),
     OCLTyp_t_exp T5 e t5 →
     OCLTyp_t_exp T5 (E_unop unop5 e) t5
 | OCLTyp_E_When : ∀ (T5:T) (e:lexp) (x:identifier) (t5:t),
     OCLTyp_t_exp T5 e t5 →
     OCLTyp_t_exp T5 (E_var x) T_bool →
     OCLTyp_t_exp T5 (E_when e x) t5
 | OCLTyp_E_Whennot : ∀ (T5:T) (e:lexp) (x:identifier) (t5:t),
     OCLTyp_t_exp T5 e t5 →
     OCLTyp_t_exp T5 (E_var x) T_bool →
     OCLTyp_t_exp T5 (E_whennot e x) t5
with OCLTyp_t_cexp : T → cexp → t → Prop :=
 | OCLTyp_Ce_E : ∀ (T5:T) (e:lexp) (t5:t),
     OCLTyp_t_exp T5 e t5 →
     OCLTyp_t_cexp T5 (Ce_exp e) t5
 | OCLTyp_Ce_If : ∀ (T5:T) (e:lexp) (ce ce':cexp) (t5:t),
     OCLTyp_t_exp T5 e T_bool →
     OCLTyp_t_cexp T5 ce t5 →
     OCLTyp_t_cexp T5 ce' t5 →
     OCLTyp_t_cexp T5 (Ce_if e ce ce') t5
 | OCLTyp_Ce_Merge : ∀ (T5:T) (x:identifier) (ce ce':cexp) (t5:t),
     OCLTyp_t_exp T5 (E_var x) T_bool →
     OCLTyp_t_cexp T5 ce t5 →
     OCLTyp_t_cexp T5 ce' t5 →
     OCLTyp_t_cexp T5 (Ce_merge x ce ce') t5.

Inductive OCLidents : T → xs → ts → Prop :=
 | OCLIdsNil : ∀ (T5:T),
     OCLidents T5 nil TS_nil
 | OCLIdsTuple : ∀ (T5:T) (x:identifier) (xs5:xs) (t5:t) (ts5:ts),
      (assoc x T5 = Some t5 ) →
     OCLidents T5 xs5 ts5 →
     OCLidents T5 (cons x xs5 ) (TS_cons t5 ts5)
with OCLt_ck : T → clock → Prop :=
 | OCLCk_base : ∀ (T5:T),
     OCLt_ck T5 Cbase
 | OCLCk_on : ∀ (T5:T) (ck:clock) (x:identifier),
     OCLt_ck T5 ck →
      (assoc x T5 = Some T_bool ) →
     OCLt_ck T5 (Con ck x)
 | OCLCk_onnot : ∀ (T5:T) (ck:clock) (x:identifier),
     OCLt_ck T5 ck →
      (assoc x T5 = Some T_bool ) →
     OCLt_ck T5 (Connot ck x)
with OCLt_eqn : T → equation → t → Prop :=
 | OCLEq_Exp : ∀ (T5:T) (y:identifier) (ck:clock) (ce:cexp) (t5:t),
     OCLt_ck T5 ck →
      (assoc y T5 = Some t5 ) →
     OCLTyp_t_cexp T5 ce t5 →
     OCLt_eqn T5 (Equation_exp y ck ce) t5
 | OCLEq_Fby : ∀ (T5:T) (y:identifier) (ck:clock) (k:constant) (e:lexp) (t5:t),
     OCLt_ck T5 ck →
      (assoc y T5 = Some t5 ) →
     OCLTyp_t_exp T5 e t5 →
     OCLTyp_t_exp T5 (E_const k) t5 →
     OCLt_eqn T5 (Equation_fby y ck k e) t5
with OCLt_eqns : T → equations → ts → Prop :=
 | OCLEqs_Nil : ∀ (T5:T),
     OCLt_eqns T5 nil TS_nil
 | OCLEqs_Cons : ∀ (T5:T) (eqn:equation) (eqns:equations) (t5:t) (ts5:ts),
     OCLt_eqn T5 eqn t5 →
     OCLt_eqns T5 eqns ts5 →
     OCLt_eqns T5 (cons eqn eqns ) (TS_cons t5 ts5)
with OCLt_node : nodedef → t → Prop :=
 | OCLNode_Def : ∀ (f:identifier) (xs5 ys:xs) (eqns:equations) (ts5 ts':ts) (T5 T':T),
      (zip ys ts' = Some T5 ) →
      (zip xs5 ts5 = Some T' ) →
     OCLt_eqns (app T5 T' ) eqns ts' →
       (names_eqns eqns ) = ys →
     OCLt_node (Node_mk f xs5 ys eqns) (T_arrow (T_types ts5) (T_types ts')).

definitions
funs Idents_ck
Fixpoint idents_ck (x1:clock) : xs:=
  match x1 with
  | Cbase ⇒ nil
  | (Con ck x) ⇒ (cons x (idents_ck ck ) )
  | (Connot ck x) ⇒ (cons x (idents_ck ck ) )
end.

definitions
funs Idents_le
Fixpoint idents_e (x1:lexp) : xs:=
  match x1 with
  | E_unit ⇒ nil
  | (E_const k) ⇒ nil
  | (E_var x) ⇒ (cons x nil )
  | (E_binop e binop5 e') ⇒ (app (idents_e e ) (idents_e e' ) )
  | (E_unop unop5 e) ⇒ (idents_e e )
  | (E_when e x) ⇒ (cons x (idents_e e ) )
  | (E_whennot e x) ⇒ (cons x (idents_e e ) )
end.

definitions
funs Idents
Fixpoint idents_ce (x1:cexp) : xs:=
  match x1 with
  | (Ce_exp e) ⇒ (idents_e e )
  | (Ce_merge x ce ce') ⇒ (cons x (app (idents_ce ce ) (idents_ce ce' ) ) )
  | (Ce_if e ce' ce'') ⇒ (app (app (idents_e e ) (idents_ce ce' ) ) (idents_ce ce'' ) )
end.

definitions
funs Idents_eqn
Fixpoint idents_eqn (x1:equation) : xs:=
  match x1 with
  | (Equation_exp y ck ce) ⇒ (idents_ce ce )
  | (Equation_fby y ck k e) ⇒ nil
end.

definitions
funs Idents_eqns
Fixpoint idents_eqns (x1:equations) : xs:=
  match x1 with
  | nil ⇒ nil
  | (cons eqn eqns ) ⇒ (app (idents_eqn eqn ) (idents_eqns eqns ) )
end.

Inductive Wseqn : xs → equation → Prop :=
 | Wsone_ce : ∀ (V:xs) (y:identifier) (ck:clock) (ce:cexp),
      (¬ In y V ) →
     In_ins (idents_ce ce ) V →
     In_ins (idents_ck ck ) V →
     Wseqn V (Equation_exp y ck ce)
 | Wsone_fby : ∀ (V:xs) (y:identifier) (ck:clock) (k:constant) (e:lexp),
      (¬ In y V ) →
     In_ins (idents_ck ck ) V →
     Wseqn V (Equation_fby y ck k e).
definitions

Inductive Wseqns : xs → equations → Prop :=
 | Wsone_eqn : ∀ (V:xs),
     Wseqns V nil
 | Wscons_ce : ∀ (V:xs) (y:identifier) (ck:clock) (ce:cexp) (eqns:equations),
     Wseqn V (Equation_exp y ck ce) →
     Wseqns (cons y V ) eqns →
     Wseqns V (cons (Equation_exp y ck ce) eqns )
 | Wscons_fby : ∀ (V:xs) (y:identifier) (ck:clock) (k:constant) (e:lexp) (eqns:equations),
     Wseqn V (Equation_fby y ck k e) →
     Wseqns (cons y V ) eqns →
     Wseqns V (cons (Equation_fby y ck k e) eqns ) .

funs Convert_lexp
Fixpoint convert_lexp_mle (x1:lexp) : mle:=
  match x1 with
  | E_unit ⇒ ML_Eunit
  | (E_const k) ⇒ (ML_Econst k)
  | (E_var x) ⇒ (ML_Evariable x)
  | (E_binop e binop5 e') ⇒ (ML_Ebinop (convert_lexp_mle e ) binop5 (convert_lexp_mle e' ) )
  | (E_when e x) ⇒ (ML_Eif (ML_Evariable x) (convert_lexp_mle e ) ML_Emagic)
  | (E_whennot e x) ⇒ (ML_Eif (ML_Evariable x) ML_Emagic (convert_lexp_mle e ) )
  | (E_unop unop5 e) ⇒ (ML_Eunop unop5 (convert_lexp_mle e ) )
end.

Fixpoint convert_cexp_mle (x1:cexp) : mle:=
