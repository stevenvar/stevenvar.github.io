Lemmes d'affaiblissement

Expressions

Lemma aff_lexp :
  ∀ e T R x' t t',
    OCamlEexpr R T (convert_lexp_mle e) t →
  OCamlEexpr ((x', t') :: R) T (convert_lexp_mle e) t.

Expressions conditionnelles

Lemma aff_ce :
  ∀ ce R T x' t' t,
 OCamlEexpr R T (convert_cexp_mle ce) t →
  OCamlEexpr ((x', t') :: R) T
    (convert_cexp_mle ce) t.

Initialisations de registres

Lemma aff_inits :
 ∀ eqns R T x t ts,
   ( not (In x (names_eqns eqns) ) ) →
 OCamlEdeclsr R T (convert_inits_decls eqns) ts →
 OCamlEdeclsr ((x, t) :: R) T (convert_inits_decls eqns) ts.

Fixpoint vars_ck ck :=
  match ck with
   | Cbase ⇒ nil
   | Con ck x ⇒ x::vars_ck ck
   | Connot ck x ⇒ x::vars_ck ck
  end.

Horloges

Lemma aff_ck :
  ∀ ck x' t' t R T,
    OCamlEexpr (R) T (convert_ck ck) t →
    OCamlEexpr ((x', t') :: R) T (convert_ck ck) t.

Equations

Lemma aff_eqns :
  ∀ eqns x' t ts R T,
    ( not (In x' (names_eqns eqns) ) ) →
  OCamlEdecls (R) T (convert_eqns_decls eqns) ts →
  OCamlEdecls ((x',t)::R) T (convert_eqns_decls eqns) ts.

Mises à jour de registres


Lemma aff_updates :
  ∀ eqns T R x t,
    not (In x (names_eqns eqns)) →
    OCamlEexpr
      R T (convert_updates eqns) T_unit
  →
    OCamlEexpr
      ((x, t) :: R) T (convert_updates eqns) T_unit.

N-uplets

Lemma aff_tuples :
  ∀ eqns T T' R ts x t0 ,
    (not (In x (names_eqns eqns))) →
  OCamlEexprs R ((x, t0) :: T ++ T')
              (convert_xs_exp (names_eqns eqns)) ts
  →
  OCamlEexprs R (T ++ T')
              (convert_xs_exp (names_eqns eqns)) ts.

Environnement
Lemma L2: ∀ T' T x i t0 t,
  assoc x (T' ++ (i, t0) :: T) = Some t →
  x ≠ i →
  assoc x (T'++T) = Some t.

Lemma aff_assoc : ∀ T x l t,
    assoc x l = None →
    assoc x (l ++ T) = Some t →
   assoc x T = Some t.

Lemma aff_ref :
  ∀ k R T x t' t,
  OCamlEexpr R T (ML_Eref (ML_Econst k)) t →
  OCamlEexpr ((x, t') :: R) T (ML_Eref (ML_Econst k)) t.

Lemma L9''' :
  ∀ T x i t0 T' t,
    x = i →
  assoc x T = None →
  assoc x (T ++ (i, t0) :: T') = Some t →
  Some t0 = Some t.

Lemma L9'' :
  ∀ T' T x t,
    assoc x T = None →
    assoc x (T ++ T') = Some t →
    assoc x (T' ++ T) = Some t.

  Lemma assoc_test : ∀ l T' T x t ,
      assoc x l = None →
      assoc x (T' ++ l ++ T) = Some t →
      assoc x (l ++ T' ++ T) = Some t.

Fixpoint vars_env (l : list (identifier × t) ) :=
  match l with
    | nil ⇒ nil
    | cons (a,_) l' ⇒ cons a (vars_env l')
  end.

Lemma not_in_vars_env_assoc :
  ∀ l x,
 ¬ In x (vars_env l) →
  assoc x l = None.

Lemma L22 :
  ∀ l T y ts,
  assoc y (l ++ T) = Some ts →
  ¬ In y (vars_env l) →
  assoc y T = Some ts.

Lemma aff_assoc_app :
  ∀ T a t0 T5,
  assoc a T = Some t0 →
  assoc a (T ++ T5) = Some t0.

Lemma aff_tuples_app :
  ∀ zs R' x t5 l ts' T5,
  OCamlEexprs R' ((x, t5) :: l) (convert_xs_exp zs) ts' →
  OCamlEexprs R' (((x, t5) :: l) ++ T5) (convert_xs_exp zs) ts'.

Expressions

OCamlEexpr ((x', t) :: R) T (convert_lexp_mle e) t' →
