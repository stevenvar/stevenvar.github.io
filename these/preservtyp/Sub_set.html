Lemmes utilitaires concernant les sous-ensembles


Lemma in_assoc' :
∀ z ys yt zt R,
  Subin z zt ys yt →
  (zip ys yt) = Some R →
  assoc z R = Some zt.

Lemma typing_preservation_subset :
  ∀ zs ys zt yt R T,
    zip ys yt = Some T →
    Subsubset zs zt ys yt →
    OCamlEexprs R T (convert_xs_exp zs) zt.

Lemma in_assoc_left :
∀ z ys yt zt R ,
  (zip ys yt) = Some R →
  assoc z R = Some zt →
  Subin z zt ys yt.

   Lemma typing_preservation_subset_left :
    ∀ zs tz ys ty R T,
    zip ys ty = Some T →
    OCamlEexprs R T (convert_xs_exp zs) tz
    →
    Subsubset zs tz ys ty.

  Lemma typing_preservation_subset_equiv :
    ∀ zs zt ys yt R T,
    zip ys yt = Some T →
    (Subsubset zs zt ys yt ↔
    OCamlEexprs R T (convert_xs_exp zs) zt).

Lemma sub_nil_zs :
  ∀ zs ts' ts'',
    Subsubset zs ts' nil ts'' → (zs = nil).

Lemma sub_nil_ts :
  ∀ zs ts ts',
    Subsubset zs ts nil ts' → (ts = TS_nil).
